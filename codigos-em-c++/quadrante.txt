

#include <iostream>
 
using namespace std;
 
int main() {
 
    int x = 1, y = 1;

    while(x != 0 || y != 0){
        cin >> x >> y;
        if(x == 0 || y == 0) break;

        //1째 Q -> x+ / y+
        if(x > 0 && y > 0) cout << "primeiro" << endl;
        //2째 Q -> x- / y+
        else if(x < 0 && y > 0) cout << "segundo" << endl;
        //3째 Q -> x- / y-
        else if(x < 0 && y < 0) cout << "terceiro" << endl;
        //4째 Q -> x+ / y-
        else if(x > 0 && y < 0) cout << "quarto" << endl;
    }
 
    return 0;
}
